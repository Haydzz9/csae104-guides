Backend — Express Server

Configure the Node.js server and database connection

Create the .env file

🔑 blog-server/.env
DB_HOST=localhost
DB_PORT=5432
DB_NAME=blog_db
DB_USER=postgres
DB_PASSWORD=your_password_here
PORT=5000

Create the database connection — db.js

⚡ blog-server/db.js JS
const { Pool } = require('pg')
require('dotenv').config()

const pool = new Pool({
  host:     process.env.DB_HOST,
  port:     process.env.DB_PORT,
  database: process.env.DB_NAME,
  user:     process.env.DB_USER,
  password: process.env.DB_PASSWORD,
})

// Test the connection on startup
pool.connect((err, client, release) => {
  if (err) return console.error('❌ DB connection error:', err.message)
  console.log('✅ Connected to PostgreSQL: blog_db')
  release()
})

module.exports = pool

Create the main server — server.js

⚡ blog-server/server.js JS
const express = require('express')
const cors    = require('cors')
require('dotenv').config()

const postsRouter = require('./routes/posts')

const app = express()

// Middleware
app.use(cors())
app.use(express.json())

// Routes
app.use('/api/posts', postsRouter)

// Health check
app.get('/', (req, res) => res.json({ message: '✅ Blog API is running!' }))

const PORT = process.env.PORT || 5000
app.listen(PORT, () => console.log(`🚀 Server running on http://localhost:${PORT}`))

API Routes — All CRUD Operations

GET, POST, PUT, DELETE endpoints for blog posts

Create the file blog-server/routes/posts.js — this file handles all 5 API endpoints:

⚡ blog-server/routes/posts.js JS
const express = require('express')
const router  = express.Router()
const pool    = require('../db')

// ── GET /api/posts ── Get all posts (newest first) ────────────
router.get('/', async (req, res) => {
  try {
    const { category } = req.query   // optional filter: ?category=React
    let sql = 'SELECT * FROM posts'
    const params = []
    if (category) {
      sql += ' WHERE category = $1'
      params.push(category)
    }
    sql += ' ORDER BY created_at DESC'
    const result = await pool.query(sql, params)
    res.json(result.rows)
  } catch (err) {
    res.status(500).json({ error: err.message })
  }
})

// ── GET /api/posts/:id ── Get one post ────────────────────────
router.get('/:id', async (req, res) => {
  try {
    const result = await pool.query(
      'SELECT * FROM posts WHERE id = $1', [req.params.id]
    )
    if (result.rows.length === 0)
      return res.status(404).json({ error: 'Post not found' })
    res.json(result.rows[0])
  } catch (err) {
    res.status(500).json({ error: err.message })
  }
})

// ── POST /api/posts ── Create a new post ──────────────────────
router.post('/', async (req, res) => {
  try {
    const { title, content, author, category } = req.body
    if (!title || !content || !author)
      return res.status(400).json({ error: 'title, content, and author are required' })

    const result = await pool.query(
      `INSERT INTO posts (title, content, author, category)
       VALUES ($1, $2, $3, $4) RETURNING *`,
      [title, content, author, category || 'General']
    )
    res.status(201).json(result.rows[0])
  } catch (err) {
    res.status(500).json({ error: err.message })
  }
})

// ── PUT /api/posts/:id ── Update a post ───────────────────────
router.put('/:id', async (req, res) => {
  try {
    const { title, content, author, category } = req.body
    const result = await pool.query(
      `UPDATE posts
       SET title=$1, content=$2, author=$3, category=$4,
           updated_at=NOW()
       WHERE id=$5 RETURNING *`,
      [title, content, author, category, req.params.id]
    )
    if (result.rows.length === 0)
      return res.status(404).json({ error: 'Post not found' })
    res.json(result.rows[0])
  } catch (err) {
    res.status(500).json({ error: err.message })
  }
})

// ── DELETE /api/posts/:id ── Delete a post ────────────────────
router.delete('/:id', async (req, res) => {
  try {
    await pool.query('DELETE FROM posts WHERE id = $1', [req.params.id])
    res.json({ message: 'Post deleted successfully' })
  } catch (err) {
    res.status(500).json({ error: err.message })
  }
})

module.exports = router

PostCard.jsx

⚛ src/components/PostCard.jsx
import React from 'react'
import { useNavigate } from 'react-router-dom'

// Gradient colours for card banners (cycles by post id)
const GRADIENTS = [
  'linear-gradient(135deg,#667eea,#764ba2)',
  'linear-gradient(135deg,#f093fb,#f5576c)',
  'linear-gradient(135deg,#4facfe,#00f2fe)',
  'linear-gradient(135deg,#43e97b,#38f9d7)',
  'linear-gradient(135deg,#fa709a,#fee140)',
]

export default function PostCard({ post }) {
  const navigate = useNavigate()
  const gradient = GRADIENTS[post.id % GRADIENTS.length]
  const date = new Date(post.created_at).toLocaleDateString('en-PH', {
    year: 'numeric', month: 'long', day: 'numeric'
  })

  return (
    <div className="post-card" onClick={() => navigate(`/posts/${post.id}`)}>
      <div className="post-card-banner"
           style={{ background: gradient }} />
      <div className="post-card-body">
        <span className="category-tag">{post.category}</span>
        <h3>{post.title}</h3>
        <p>By {post.author} · {date}</p>
      </div>
    </div>
  )
}

PostForm.jsx — Reusable for Create & Edit

⚛ src/components/PostForm.jsx
import React, { useState } from 'react'

const CATEGORIES = ['General','React','Node.js','Database','CSS','JavaScript']

export default function PostForm({ initial, onSubmit, loading }) {
  const [form, setForm] = useState(initial || {
    title: '', content: '', author: '', category: 'General'
  })

  const handleChange = e =>
    setForm(prev => ({ ...prev, [e.target.name]: e.target.value }))

  const handleSubmit = e => {
    e.preventDefault()
    onSubmit(form)
  }

  return (
    <form className="form-card" onSubmit={handleSubmit}>
      <div className="form-group">
        <label>Title</label>
        <input name="title" value={form.title}
               onChange={handleChange} required />
      </div>
      <div className="form-group">
        <label>Author</label>
        <input name="author" value={form.author}
               onChange={handleChange} required />
      </div>
      <div className="form-group">
        <label>Category</label>
        <select name="category" value={form.category} onChange={handleChange}>
          {CATEGORIES.map(c => <option key={c}>{c}</option>)}
        </select>
      </div>
      <div className="form-group">
        <label>Content</label>
        <textarea name="content" value={form.content}
                  onChange={handleChange} required />
      </div>
      <button className="btn btn-primary" type="submit" disabled={loading}>
        {loading ? 'Saving…' : 'Save Post'}
      </button>
    </form>
  )
}

Home.jsx — List all posts

⚛ src/pages/Home.jsx
import React, { useState, useEffect } from 'react'
import api from '../api'
import PostCard from '../components/PostCard'

const CATS = ['All','React','Node.js','Database','CSS','JavaScript']

export default function Home() {
  const [posts,    setPosts]    = useState([])
  const [loading,  setLoading]  = useState(true)
  const [category, setCategory] = useState('All')

  useEffect(() => {
    const params = category !== 'All' ? { category } : {}
    api.get('/posts', { params })
      .then(res => { setPosts(res.data); setLoading(false) })
      .catch(err => console.error(err))
  }, [category])

  return (
    <div className="container">
      <div style={{ display:'flex', gap:'.6rem', marginBottom:'2rem', flexWrap:'wrap' }}>
        {CATS.map(c => (
          <button key={c}
            className={`btn ${category===c ? 'btn-primary' : 'btn-ghost'}`}
            onClick={() => setCategory(c)}>{c}</button>
        ))}
      </div>
      {loading
        ? <p>Loading posts…</p>
        : <div className="post-grid">
            {posts.map(p => <PostCard key={p.id} post={p} />)}
          </div>
      }
    </div>
  )
}

PostDetail.jsx — View + Delete

⚛ src/pages/PostDetail.jsx
import React, { useState, useEffect } from 'react'
import { useParams, useNavigate, Link } from 'react-router-dom'
import api from '../api'

export default function PostDetail() {
  const { id }     = useParams()
  const navigate  = useNavigate()
  const [post, setPost]       = useState(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    api.get(`/posts/${id}`)
      .then(res => { setPost(res.data); setLoading(false) })
      .catch(() => navigate('/'))
  }, [id])

  const handleDelete = async () => {
    if (!window.confirm('Delete this post?')) return
    await api.delete(`/posts/${id}`)
    navigate('/')
  }

  if (loading) return <p className="container">Loading…</p>

  return (
    <div className="container" style={{maxWidth:'720px'}}>
      <span className="category-tag">{post.category}</span>
      <h1 style={{margin:'1rem 0 .5rem',lineHeight:1.3}}>{post.title}</h1>
      <p style={{color:'#6e7781',marginBottom:'1.5rem'}}>
        By {post.author} · {new Date(post.created_at).toLocaleDateString()}
      </p>
      <p style={{lineHeight:1.8,marginBottom:'2rem'}}>{post.content}</p>
      <div style={{display:'flex',gap:'.8rem'}}>
        <Link to={`/edit/${id}`} className="btn btn-ghost">Edit Post</Link>
        <button className="btn btn-danger" onClick={handleDelete}>Delete</button>
        <Link to="/" className="btn btn-ghost">← Back</Link>
      </div>
    </div>
  )
}

import React, { useState, useEffect } from 'react'
